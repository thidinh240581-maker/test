local function GetFruitName(plr)
    if plr.Character then
        for _, tool in pairs(plr.Character:GetChildren()) do
            if tool:IsA("Tool") and tool:FindFirstChild("Fruit") then
                return tool.Name
            end
        end
    end
    return nil
end

local function IsV4Active(plr)
    return plr.Character and (plr.Character:FindFirstChild("RaceTransformed") or plr.Character:FindFirstChild("Dragon") or plr.Character:FindFirstChild("Leopard"))
end

local function GetNearestPlayer()
    local nearest, dist = nil, math.huge
    for _, v in ipairs(Players:GetPlayers()) do
        if v ~= player and v.Character and v.Character:FindFirstChild("HumanoidRootPart") and v.Character.Humanoid.Health > 0 then
            
            -- Skip PvP Disabled
            if v.Character:FindFirstChild("PvPDisabled") or v.Character.Humanoid.Health == 0 then continue end

            if Config.Skip.Fruit.Enabled then
                local fruit = GetFruitName(v)
                if fruit and table.find(Config.Skip.Fruit.AvoidFruit, fruit) then continue end
            end
            if Config.Skip.AvoidV4 and IsV4Active(v) then continue end

            local d = (root.Position - v.Character.HumanoidRootPart.Position).Magnitude
            if d < dist then
                dist = d
                nearest = v
            end
        end
    end
    return nearest
end

-- Orbit
local function StartOrbit(target)
    if OrbitConnection then OrbitConnection:Disconnect() end
    OrbitConnection = RunService.Heartbeat:Connect(function()
        if not target or not target.Character or not root then return end
        local angle = tick() * Config.Other.SpinBot.SpinSpeed
        local offset = Vector3.new(math.cos(angle), 0, math.sin(angle)) * Config.Other.SpinBot.SpinRadius
        root.CFrame = CFrame.new(target.Character.HumanoidRootPart.Position + offset, target.Character.HumanoidRootPart.Position)
    end)
end

local function StopOrbit()
    if OrbitConnection then OrbitConnection:Disconnect() OrbitConnection = nil end
end

-- Skill System
local function UseSkill(weapon, skill)
    local data = Config.Items[weapon] and Config.Items[weapon].Skills[skill]
    if not data or not data.Enable then return end

    local tool = character:FindFirstChild(weapon) or character:FindFirstChildOfClass("Tool")
    if tool then
        tool:Activate()
        if data.HoldTime > 0 then
            task.wait(data.HoldTime)
            VirtualUser:Button1Up(Vector2.new(0,0), Workspace.CurrentCamera.CFrame)
        end
    end
end

local function AttackTarget(target)
    if not target or not target.Character then return end

    if IsV4 and Config.SpamAllSkillOnV4.Enabled then
        for _, wp in ipairs(Config.SpamAllSkillOnV4.Weapons) do
            for sk in pairs(Config.Items[wp].Skills) do
                UseSkill(wp, sk)
            end
        end
        return
    end

    for _, weapon in ipairs(Config.Items.Use) do
        if Config.Items[weapon].Enable then
            for sk, data in pairs(Config.Items[weapon].Skills) do
                if data.Enable then
                    UseSkill(weapon, sk)
                    task.wait(Config.Other.ClickDelay or 0.3)
                end
            end
        end
    end
end

-- Main Loop
task.spawn(function()
    while task.wait(0.1) do
        pcall(function()
            Target = GetNearestPlayer()
            if Target then
                if Config.Other.SpinBot.Enable then
                    StartOrbit(Target)
                else
                    StopOrbit()
                end
                AttackTarget(Target)
            else
                StopOrbit()
            end
        end)
    end
end)

local function TweenTo(pos, speed)
    if Tween then Tween:Cancel() end
    if not root then return end
    local dist = (root.Position - pos.Position).Magnitude
    Tween = TweenService:Create(root, TweenInfo.new(dist / (speed or 350), Enum.EasingStyle.Linear), {CFrame = pos})
    Tween:Play()
    Tween.Completed:Wait()
end
